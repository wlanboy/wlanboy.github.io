Gist: create-aarch64-kvm-guest

sudo apt-get install qemu-system-arm qemu-efi-aarch64 qemu-utils libarchive-tools

qemu-img create -f qcow2 debian-aarch64.qcow2 5G
wget https://cdimage.debian.org/debian-cd/current/arm64/iso-cd/debian-10.3.0-arm64-netinst.iso

#--os-variant=debian10 still not available
virt-install --os-type=linux --os-variant=debian9 --arch aarch64 --name debian-aarch64 --ram=2048 --vcpus=1 \
--disk path=./debian-aarch64.qcow2,bus=virtio,format=qcow2 \
--location ./debian-10.3.0-arm64-netinst.iso \
--network network=br0,model=virtio
