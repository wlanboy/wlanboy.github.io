Gist: install-golang-on-arm

wget https://dl.google.com/go/go1.13.5.linux-armv6l.tar.gz
tar -xvf go1.13.5.linux-armv6l.tar.gz

echo "export PATH=$PATH:~/go/bin" >> ~/.bashrc
mkdir ~/gowork
echo "export GOPATH=~/gowork" >> ~/.bashrc
